  // Extend the default translations for new languages by copying the English set.
  [ 'de', 'fr', 'es', 'it', 'nl', 'pt', 'da', 'no', 'fi', 'pl' ].forEach(function (lang) {
    translations[lang] = Object.assign({}, translations['en']);
  });

  /**
   * Retrieve the user\'s preferred language from localStorage.
   * Defaults to English if not set.
   * @returns {string} The language code.
   */
  function getCurrentLang() {
    return localStorage.getItem('preferredLang') || 'en';
  }

  // Expose getCurrentLang on window so other scripts can access it
  window.getCurrentLang = getCurrentLang;

  /**
   * Set the page language. This updates all elements with a
   * data-i18n attribute, swaps Bootstrap RTL/LTR CSS, and stores
   * the preference in localStorage.
   * @param {string} lang The language code to set.
   */
  function setLanguage(lang) {
    if (!translations[lang]) lang = 'en';
    // update all elements
    document.querySelectorAll('[data-i18n]').forEach((el) => {
      const key = el.getAttribute('data-i18n');
      const text = translations[lang][key];
      if (text !== undefined) {
        el.innerHTML = text;
      }
    });
    // Update value attributes on elements that have data-i18n-value
    document.querySelectorAll('[data-i18n-value]').forEach((el) => {
      const key = el.getAttribute('data-i18n-value');
      const text = translations[lang][key];
      if (text !== undefined) {
        el.value = text;
      }
    });
    // update html attributes for direction and language
    document.documentElement.lang = lang;
    if (lang === 'ar') {
      document.documentElement.dir = 'rtl';
      // swap bootstrap to RTL version
      const css = document.getElementById('bootstrap-css');
      if (css) {
        css.href = 'https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/css/bootstrap.rtl.min.css';
      }
    } else {
      document.documentElement.dir = 'ltr';
      const css = document.getElementById('bootstrap-css');
      if (css) {
        css.href = 'https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/css/bootstrap.min.css';
      }
    }
    // update select
    const selector = document.getElementById('language-select');
    if (selector) selector.value = lang;
    // store preference
    localStorage.setItem('preferredLang', lang);
    // update document title if translation provided
    const titleKey = {
      'index.html': 'nav-home',
      'about.html': 'about-title',
      'product.html': 'product-title',
      'contact.html': 'contact-title'
    };
    const page = location.pathname.split('/').pop();
    const pageKey = titleKey[page];
    if (pageKey && translations[lang][pageKey]) {
      document.title = translations[lang][pageKey];
    }

    // Update cart display if the function exists to refresh totals and labels in the selected language
    if (typeof updateCartDisplay === 'function') {
      updateCartDisplay();
    }
  }

  /**
   * Handler for the language dropdown. Simply calls setLanguage.
   * @param {string} lang The selected language.
   */
  window.changeLang = function (lang) {
    setLanguage(lang);
  };

  /**
   * Exported function to get a translation string in the current language.
   * Useful in other scripts like cart handling.
   * @param {string} key The translation key.
   * @returns {string} The translated string.
   */
  window.getTranslation = function (key) {
    const lang = getCurrentLang();
    return translations[lang] && translations[lang][key] ? translations[lang][key] : translations['en'][key] || '';
  };

  // On DOM ready, set initial language
  document.addEventListener('DOMContentLoaded', () => {
    const lang = getCurrentLang();
    setLanguage(lang);
  });
})();

// cart management using localStorage
/**
 * Update the cart count badge and offcanvas display. If the offcanvas
 * is present on the page, this function will also populate the list
 * of items and the total price. Cart count is stored in localStorage
 * under the key `cartCount`.
 */
function loadCartCount() {
  updateCartDisplay();
}

function addToCart() {
  try {
    var count = parseInt(localStorage.getItem('cartCount')) || 0;
    count += 1;
    localStorage.setItem('cartCount', count);
    updateCartDisplay();
    // Show toast notification instead of alert
    var toastEl = document.getElementById('cartToast');
    if (toastEl) {
      var msgEl = toastEl.querySelector('.toast-body');
      if (msgEl && typeof getTranslation === 'function') {
        msgEl.innerHTML = getTranslation('cart-added');
      }
      var toast = bootstrap.Toast.getOrCreateInstance(toastEl);
      toast.show();
    }
  } catch (e) {
    console.error('Cart update error:', e);
  }
}

/**
 * Update cart badge and offcanvas contents. Reads cartCount from
 * localStorage, calculates total based on fixed price, and inserts
 * translated text into the offcanvas.
 */
function updateCartDisplay() {
  try {
    var count = parseInt(localStorage.getItem('cartCount')) || 0;
    // Update badge
    var badge = document.getElementById('cart-count');
    if (badge) {
      badge.textContent = count;
    }
    // Update offcanvas items and total if available
    var itemsDiv = document.getElementById('cart-items');
    var totalSpan = document.getElementById('cart-total');
    var titleEl = document.getElementById('cartOffcanvasLabel');
    if (titleEl && typeof getTranslation === 'function') {
      titleEl.innerHTML = getTranslation('cart-title');
    }
    if (itemsDiv) {
      if (count === 0) {
        if (typeof getTranslation === 'function') {
          itemsDiv.innerHTML = '<p>' + getTranslation('cart-empty') + '</p>';
        } else {
          itemsDiv.innerHTML = '<p>Your cart is empty.</p>';
        }
      } else {
        // Build item line
        var line = '<p>Padel Racket × ' + count + '</p>';
        itemsDiv.innerHTML = line;
      }
    }
    if (totalSpan) {
      var price = 799.99;
      var total = count * price;
      // Format number with two decimals and comma/point according to language
      var lang = (typeof getCurrentLang === 'function') ? getCurrentLang() : 'en';
      var formatted;
      if (lang === 'sv') {
        // Use Swedish locale formatting for decimal comma
        try {
          formatted = total.toLocaleString('sv-SE', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
        } catch (e) {
          formatted = total.toFixed(2).replace('.', ',');
        }
      } else if (lang === 'ar') {
        // Arabic also uses comma as decimal separator
        formatted = total.toFixed(2).replace('.', ',');
      } else {
        formatted = total.toFixed(2);
      }
      totalSpan.textContent = formatted;
    }
  } catch (e) {
    console.error('updateCartDisplay error:', e);
  }
}

/**
 * Open the cart offcanvas programmatically. This helper updates
 * the cart display then shows the offcanvas using Bootstrap's JS API.
 */
function openCart() {
  try {
    updateCartDisplay();
    var offcanvasEl = document.getElementById('cartOffcanvas');
    if (offcanvasEl) {
      var offcanvas = bootstrap.Offcanvas.getOrCreateInstance(offcanvasEl);
      offcanvas.show();
    }
  } catch (e) {
    console.error('openCart error:', e);
  }
}
